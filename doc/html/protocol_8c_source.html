607 /* Subtract the `struct timespec' values X and Y,
608  storing the result in RESULT.
609  Return 1 if the difference is negative, otherwise 0. */
610 
611 int timespec_subtract (result, x, y) struct timespec *result, *x, *y;
612  {
613  /* Perform the carry for the later subtraction by updating y. */
614  if (x->tv_nsec < y->tv_nsec) {
615  int nsec = (y->tv_nsec - x->tv_nsec) / 1000000000 + 1;
616  y->tv_nsec -= 1000000000 * nsec;
617  y->tv_sec += nsec;
618  }
619  if (x->tv_nsec - y->tv_nsec > 1000000000) {
620  int nsec = (x->tv_nsec - y->tv_nsec) / 1000000000;
621  y->tv_nsec += 1000000000 * nsec;
622  y->tv_sec -= nsec;
623  }
624 
625  /* Compute the time remaining to wait.
626  tv_nsec is certainly positive. */
627  result->tv_sec = x->tv_sec - y->tv_sec;
628  result->tv_nsec = x->tv_nsec - y->tv_nsec;
629 
630  /* Return 1 if result is negative. */
631  return x->tv_sec < y->tv_sec;
632  }
